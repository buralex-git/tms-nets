147  Polynom
148  multiply_poly2(
150  Polynom const &poly_pa,
152  Polynom const &poly_pb)
153  {
154  std::size_t jlo, jhi;
155  BasicInt term;
156  Polynom poly_pc(std::vector< irrpoly::gf<2> >(poly_pa.degree() + poly_pb.size(), 1));
157 
158  for (std::size_t i = 0; i <= poly_pa.degree() + poly_pb.degree(); ++i)
159  {
160  jlo = ( i < poly_pa.degree() ) ? 0 : static_cast<BasicInt>(i - poly_pa.degree());
161  jhi = ( i < poly_pb.degree() ) ? i : static_cast<BasicInt>(poly_pb.degree());
162  //
163  // Find pc_i as a mod2 sum over all pa_j*pb_k such that j + k = i
164  //
165  term = 0;
166  for (std::size_t j = jlo; j <= jhi; ++j)
167  {
168  term ^= poly_pa[i - j].data() & poly_pb[j].data();
169  }
170  poly_pc[i] = term;
171  }
172 
173  return poly_pc;
174  }

215  for (BasicInt i = 1; index >> i; ++i, ++degree);

357  std::vector<Polynom>
358  generate_irrpolys_with_degrees(
360  std::vector<BasicInt> const &degrees,
362  BasicInt const max_defect)
363  {
364  BasicInt amount = static_cast<BasicInt>(degrees.size());
365  // counter of possible t values for (t,m,s)-nets with sush irred. polynomials degrees.
366  BasicInt defect = 0;
367  // coefficient number of polynomial over GF(2) such that n-th bit is equal to n-th coefficient of polynomial.
368 
369  auto number_to_poly = [](CountInt coeffs_number) {
370  Polynom poly;
371  BasicInt degree = 0;
372  while ( coeffs_number >> degree != 0 ) { ++degree; }
373  poly.data().reserve(degree--);
374  for (BasicInt i = 0; i <= degree; ++i)
375  {
376  poly.data().emplace_back((coeffs_number >> i) & 1);
377  }
378  return poly;
379  };
380 
381  // map of elements { key, value }:
382  // key == degree;
383  // value == {degree's appearances count in degrees, current coefficient number } },
384  // where degree is taken from degrees vector.
385  std::map< BasicInt, std::pair<BasicInt, BasicInt> > degrees_counter;
386 
387  BasicInt i = 0;
388  for (i = 0; i < amount && defect <= max_defect && degrees[i] != 0; ++i)
389  {
390  // if there's no key == degree, it will be added with value 0 and then will be incremented to 1;
391  // else the value associated with key == degree will be just incremented.
392  degrees_counter[degrees[i]].first += 1;
393  degrees_counter[degrees[i]].second = (1 << degrees[i]) + (degrees[i] != 1);
394  defect += degrees[i] - 1;
395  }
396 
397  if ( i < amount && degrees[i] == 0 )
398  {
399  throw std::logic_error("\nThere is no polynomial with degree 0");
400  }
401 
402  amount = i - ( defect > max_defect );
403  std::vector<Polynom> irred_polys_table;
404  irred_polys_table.reserve(amount);
405 
406  for (i = 0; i < amount && degrees_counter[degrees[i]].second < (2U << degrees[i]); ++i)
407  {
408  irred_polys_table.emplace_back( number_to_poly(degrees_counter[degrees[i]].second) );
409 
410  while ( degrees_counter[degrees[i]].second < (2U << degrees[i]) && !is_irreducible_berlekamp(irred_polys_table.back()) )
411  {
412  degrees_counter[degrees[i]].second += 2;
413  irred_polys_table.back() = number_to_poly(degrees_counter[degrees[i]].second);
414  }
415  degrees_counter[degrees[i]].second += (degrees[i] != 1) + 1;
416  }
417 
418  if ( i < amount && degrees_counter[degrees[i]].second >= (2U << degrees[i]) )
419  {
420  throw std::logic_error("\nThere are no " + std::to_string(degrees_counter[degrees[i]].first) + " irreducible polynomials of degree " + std::to_string(degrees[i]) + " over GF(2)");
421  }
422 
423  return irred_polys_table;
424  }

601  template <typename UIntType, unsigned int NBITS>
602  void
603  Niederreiter<UIntType, NBITS>::calculate_v(
605  Polynom const &poly_pi,
610  Polynom &poly_b,
612  std::vector<BasicInt> &v) const
613  {
614  Polynom poly_h;
615  BasicInt term;
616  //
617  // The polynomial h(x) = p_i(x)**(j-1) = b(x) on arrival.
618  //
619  // In section 3.3, the values of h_i are defined with a minus sign,
620  // but in GF(2): h_i = -h_i for any value.
621  //
622  poly_h = poly_b;
623  //
624  // Now choose a value of K_j as defined in section 3.3.
625  // We must have 0 <= K_j < e*j = m.
626  // The limit condition on K_j does not seem very relevant
627  // in this program.
628  // Let K_j = e*(j - 1) = deg(h).
629  //
630  // Multiply b(x) by p_i(x) so b(x) becomes p(x)**j.
631  // In section 2.3, the values of b_i are defined with a minus sign,
632  // but in GF(2): b_i = -b_i for any value.
633  //
634  poly_b = multiply_poly2(poly_pi, poly_b);
635  //
636  // Choose values of {v_n} in accordance with the conditions in section 3.3.
637  //
638  for (BasicInt r = 0; r < poly_h.degree(); ++r)
639  {
640  v[r] = 0;
641  }
642  v[poly_h.degree()] = 1;
643 
644  for (BasicInt r = static_cast<BasicInt>(poly_h.degree() + 1); r <= poly_b.degree() - 1; ++r)
645  {
646  v[r] = 1;
647  }
648  //
649  // Calculate the remaining v_n's using the recursion of section 2.3,
650  // remembering that the b_i's have the opposite sign.
651  //
652  for (BasicInt r = 0; r < v.size() - poly_b.degree(); ++r)
653  {
654  term = 0;
655  for (BasicInt i = 0; i <= poly_b.degree() - 1; ++i)
656  {
657  term ^= poly_b[i].data() & v[r + i];
658  }
659  v[r + poly_b.degree()] = term;
660  }
661 
662  return;
663  }

696  template <typename UIntType, unsigned int NBITS>
697  void
698  Niederreiter<UIntType, NBITS>::initialize_c(void)
699  {
700  Polynom const poly_ident{1};
701  Polynom poly_b;
702  Polynom poly_pi;
703  std::vector<BasicInt> v(NBITS + \
704  std::max_element(
705  c_irred_polys.begin(), \
706  c_irred_polys.end(), \
707  [](Polynom const &lpoly, Polynom const &rpoly) { return lpoly.degree() < rpoly.degree(); } \
708  )->degree() + 1);
709 
710  for (BasicInt i = 0, u = 0; i < c_dim; ++i)
711  {
712  //
713  // For each dimension, we need to calculate powers of an
714  // appropriate irreducible polynomial: see Niederreiter
715  // page 65, just below equation (19).
716  //
717  // Copy the appropriate irreducible polynomial p_i(x) into pi,
718  // and its degree into e. Set polynomial b(x) = 1 and copy it into b.
719  // m is the degree of b(x). Subsequently b(x) will hold higher
720  // powers of p_i(x).
721  //
722  poly_pi = c_irred_polys[i];
723  poly_b = poly_ident;
724  //
725  // Niederreiter (page 56, after equation (7), defines two
726  // variables q(i, j) and u(i, j). We do not need q explicitly, but we do need u.
727  //
728  u = 0;
729 
730  for (BasicInt j = 0; j < NBITS; ++j)
731  {
732  //
733  // If u = 0, we need to set B to the next power of p_i(x)
734  // and recalculate {v_n}. This is done by subroutine CALCV2.
735  //
736  if ( u == 0 )
737  {
738  calculate_v(poly_pi, poly_b, v);
739  }
740  //
741  // Now c is obtained from v.
742  // !!! We can think about c(i0,j0,r) as a (NBITS - j0 - 1)-th bit in binary
743  // representation of a number cj(i0, r).
744  //

875  void
876  Niederreiter<UIntType, NBITS>::store_next_point_int(
878  IntPoint &point,

882  IntPoint const &prev_point) const
883  {
884  BasicInt rightmost_zero_bit_pos = 0;
885  pos -= (pos != 0);
886  // here we get pos = pow(2, rightmost zero bit position in pos)
887  pos = ~pos & (pos + 1);
